<body>

    <script src="https://cdnjs.cloudflare.com/ajax/libs/pixi.js/5.3.10/pixi.min.js"></script>

    <script>

        // Constants

        const WIDTH = 800;

        const HEIGHT = 600;


        // Create PIXI Application

        const app = new PIXI.Application({ width: WIDTH, height: HEIGHT });

        document.body.appendChild(app.view);


        // Create cancer cells

        const initialCellCount = 10;

        const cells = [];

        for (let i = 0; i < initialCellCount; i++) {

            const cell = new PIXI.Graphics();

            cell.beginFill(0xFF0000); // Red color

            cell.drawCircle(0, 0, 10);

            cell.endFill();

            cell.x = Math.random() * WIDTH;

            cell.y = Math.random() * HEIGHT;

            app.stage.addChild(cell);

            cells.push(cell);

        }


        // Main game loop

        app.ticker.add(() => {

            // Update

            cells.forEach(cell => {

                cell.x += (Math.random() - 0.5) * 5; // Random movement

                cell.y += (Math.random() - 0.5) * 5;

            });

        });

    </script>
